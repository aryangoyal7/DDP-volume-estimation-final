\begin{frame}{Why we added it}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{\color{ink}The problem}
      \begin{itemize}
        \item Earlier the app measured the \emph{whole frame} only --- no way to
              isolate one object in a cluttered scene.
        \item Manual masks (clicks / boxes) are slow and need a trained operator.
      \end{itemize}
      \vspace{6pt}
      \textbf{\color{ink}The goal}
      \begin{itemize}
        \item Let \emph{anyone} pick the target in plain English.
        \item Measure only the object of interest, not the background.
        \item Keep the accurate depth pipeline that already worked best.
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{What changed}
        \centering\vspace{2pt}
        {\color{slate}From}\\[2pt]
        \chip{Full-frame only}\\[10pt]
        {\large\color{indigo}$\Downarrow$}\\[8pt]
        {\color{slate}To}\\[2pt]
        \chip{Type ``waste pile'' $\rightarrow$ measured}
        \vspace{2pt}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Under the hood}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{\color{ink}Pipeline}
      \begin{enumerate}
        \item \textbf{Text $\rightarrow$ mask:} LangSAM
              (GroundingDINO + SAM~2.1) turns the prompt into a pixel mask.
        \item \textbf{Depth:} Depth Anything~V2 (Metric Indoor) --- the model
              that gave our best results.
        \item \textbf{Height + volume:} floor-plane fit, calibrated scale,
              integrate height inside the mask only.
      \end{enumerate}
      \vspace{4pt}
      {\color{slate}\small Output: calibrated volume (L), mass (kg), plus
       depth map, height heatmap and the selection mask.}
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering
      \includegraphics[width=\linewidth,height=0.74\textheight,keepaspectratio]%
        {images/portal_results.png}\\[2pt]

       end-to-end.}
    \end{column}
  \end{columns}
\end{frame}
